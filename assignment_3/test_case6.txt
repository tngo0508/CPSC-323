@ convert  (month : integer)
{
           return month / 12;
}
%%
     integer  month;
	 floating age, remainder;
	 read (month);
	 write(convert [month]);
     age := age + remainder;
	 
	 if(age < 1) 
		write(Baby);
fi
	 if  (age < 3)
		write (Toddler);
fi
	 if  (age < 5)
		write (Preschool);
fi
	 if  (age < 12)
		write (GradeSchool);
fi
	 if  (age < 18)
	   write (Teen);
fi
	 if	 (age < 21)
	    write (YoungAdult);
fi
	 else (age => 21)
	    write (Adult); 
fi
